    idusuario SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
);

-- Tabla 'clientes'
CREATE TABLE clientes (
    idcli INT PRIMARY KEY AUTO_INCREMENT,               -- Identificador único para el cliente
    Pnombre VARCHAR(50) NOT NULL,           -- Primer nombre
    Snombre VARCHAR(50),                    -- Segundo nombre
    Papellido VARCHAR(50) NOT NULL,         -- Primer apellido
    Sapellido VARCHAR(50),                  -- Segundo apellido
    DNI VARCHAR(20) UNIQUE NOT NULL,        -- Documento Nacional de Identidad (DNI)
    Telefono VARCHAR(20),                   -- Teléfono
    Email VARCHAR(100) UNIQUE,              -- Email del cliente
    Genero CHAR(1) CHECK (Genero IN ('M', 'F')), -- Género (M/F)
    FechaNacimiento DATE                    -- Fecha de nacimiento
);


-- Asegúrate de que el DNI y el Email sean únicos en la tabla 'clientes'
CREATE UNIQUE INDEX idx_clientes_dni ON clientes(DNI);
CREATE UNIQUE INDEX idx_clientes_email ON clientes(Email);

CREATE TABLE provincias (
    id_provincia INT PRIMARY KEY AUTO_INCREMENT,
    codigo varchar(2) not null,
    provincia VARCHAR(100) NOT NULL
);
CREATE TABLE cantones (
    id_canton INT PRIMARY KEY AUTO_INCREMENT,
    codigo varchar(2) not null,
    canton VARCHAR(100) NOT NULL,
    id_provincia INT,
    FOREIGN KEY (id_provincia) REFERENCES provincias(id_provincia)
);

CREATE TABLE parroquias (
    id_parroquia INT PRIMARY KEY AUTO_INCREMENT,
    codigo varchar(2) not null,
    parroquia VARCHAR(100) NOT NULL,
    id_canton INT,
    FOREIGN KEY (id_canton) REFERENCES cantones(id_canton)
);

-- Tabla 'direcciones'
CREATE TABLE direcciones (
    iddir INT PRIMARY KEY AUTO_INCREMENT,               -- Identificador único para la dirección
    idcli int,   -- Relación con el cliente
    id_provincia int,        -- Provincia
    id_canton int ,           -- Cantón
    id_parroquia int,        -- Parroquia
    CallePrincipal VARCHAR(255) NOT NULL,   -- Calle principal
    CalleSecundaria VARCHAR(255),           -- Calle secundaria
    Numero VARCHAR(10),                     -- Número de casa/departamento
    Lugar VARCHAR(100),                     -- Lugar adicional de referencia
    CodigoPostal INTEGER,
    FOREIGN KEY (idcli) REFERENCES clientes(idcli),
    FOREIGN KEY (id_provincia) REFERENCES provincias(id_provincia),
    FOREIGN KEY (id_canton) REFERENCES cantones(id_canton),
    FOREIGN KEY (id_parroquia) REFERENCES parroquias(id_parroquia)                
);
